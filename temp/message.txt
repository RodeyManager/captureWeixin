	var that = this;
	if(that.outh)that.outh=clearTimeout(that.outh);
	if(!that.on){
		that.on=true;
		this.overh=setTimeout(function(){
			//if(typeof console!="undefined"){console.log("inmouseover"+(qrcodenum++))}
			that.opacity=0;
			that.right=0;
			that.imginit();
			if(that.box2){
				addClass(that.box2,"on");
			}else{
				addClass(that.box,"on");
				addClass(that.box,"hover");
			}
			var starttime = new Date().getTime();
			function movefunc(){
				var movetime = new Date().getTime();
				if(that.move(movetime-starttime)){
					that.moveh = setTimeout(function(){movefunc()},50);
				}
			}
			movefunc();
		},10);
		qrcodelog("mouseover");
	}
}
qrcode.prototype.mouseout=function(e){
	var ele = e.relatedTarget;
	if (!ele||e.toElement) {
        ele = e.toElement;
    }
	if(ele&&(contains(this.box,ele)||ele==this.box))return;
	var that = this;
	this.outh=setTimeout(function(){
		//if(typeof console!="undefined"){console.log("inmouseout"+(qrcodenum++))}
		if(that.overh)clearTimeout(that.overh);
		if(that.moveh)clearTimeout(that.moveh);
		that.on=false;
		if(that.box2){
			removeClass(that.box2,"on");
		}else{
			removeClass(that.box,"on");
			removeClass(that.box,"hover");
		}
	},10);
	qrcodelog("mouseout");
}
